# Команды для копирования файлов бэкенда на сервер
# Выполните эти команды в НОВОМ терминале на вашей локальной машине

# 1. Копирование файлов на сервер
scp backend/routes/livekit.ts root@cv5581763:/root/livi-app/backend/routes/livekit.ts
scp backend/index.ts root@cv5581763:/root/livi-app/backend/index.ts

# Или если путь другой, замените /root/livi-app/backend на ваш путь
# Чтобы узнать путь, на сервере выполните: pwd

# 2. После копирования, на сервере (где вы сейчас) выполните:
cd /root/livi-app/backend  # или ваш путь к бэкенду
npm run start

# Или если используете PM2:
# pm2 restart backend

# Или если используете systemd:
# sudo systemctl restart backend
