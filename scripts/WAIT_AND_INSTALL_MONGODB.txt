===========================================
ОЖИДАНИЕ И УСТАНОВКА MONGODB
===========================================

В SSH СЕССИИ НА СЕРВЕРЕ ВЫПОЛНИТЕ:

# 1. Проверить процессы apt
ps aux | grep apt-get

# 2. Подождать 30 секунд (чтобы apt завершился)
echo "Ожидание завершения apt..."
sleep 30

# 3. Проверить, завершился ли процесс
ps aux | grep apt-get | grep -v grep

# 4. Если процесс все еще работает, завершить его
# killall apt-get  # Только если процесс завис!

# 5. Повторить apt update
apt update

# 6. Установить MongoDB
apt install -y mongodb-org

# 7. Запустить MongoDB
systemctl start mongod
systemctl enable mongod

# 8. Проверить статус MongoDB
systemctl status mongod

# 9. Перезапустить backend
pm2 restart livi-backend

# 10. Проверить логи
pm2 logs livi-backend --lines 20

# 11. Проверить API
curl http://localhost:3000/api/turn-credentials
