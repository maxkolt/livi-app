===========================================
ИСПРАВЛЕНИЕ БЛОКИРОВКИ APT И УСТАНОВКА MONGODB
===========================================

В SSH СЕССИИ НА СЕРВЕРЕ ВЫПОЛНИТЕ:

# 1. Проверить, какой процесс держит lock
ps aux | grep 125096

# 2. Если процесс завис, подождать немного или убить его
# (Обычно лучше подождать, но если завис - можно убить)
# kill 125096  # Только если процесс действительно завис!

# 3. Подождать немного и повторить apt update
sleep 5
apt update

# 4. Установить MongoDB
apt install -y mongodb-org

# 5. Запустить MongoDB
systemctl start mongod

# 6. Включить автозапуск MongoDB
systemctl enable mongod

# 7. Проверить статус MongoDB
systemctl status mongod

# 8. Перезапустить backend
pm2 restart livi-backend

# 9. Проверить логи backend
pm2 logs livi-backend --lines 20

# 10. Проверить работу API
curl http://localhost:3000/api/turn-credentials
