===========================================
ЗАВЕРШЕНИЕ ЗАВИСШЕГО APT И УСТАНОВКА MONGODB
===========================================

В SSH СЕССИИ НА СЕРВЕРЕ ВЫПОЛНИТЕ:

# 1. Проверить статус процесса apt
ps aux | grep 125096

# 2. Завершить зависший процесс apt (если он действительно завис)
kill 125096

# 3. Подождать немного
sleep 3

# 4. Повторить apt update
apt update

# 5. Установить MongoDB
apt install -y mongodb-org

# 6. Запустить MongoDB
systemctl start mongod
systemctl enable mongod

# 7. Проверить статус MongoDB
systemctl status mongod

# 8. Перезапустить backend
pm2 restart livi-backend

# 9. Проверить логи
pm2 logs livi-backend --lines 20

# 10. Проверить API
curl http://localhost:3000/api/turn-credentials
